NAME = minishell

CFILES = src/minishell.c \
src/validation.c \
src/builtins/builtin.c \
src/execution/execution.c \
src/execution/traverse.c \
src/execution/one_pass.c \
src/execution/two_pass.c \
src/execution/commands.c \
src/parsing/ast.c \

utils/exec_frees.c \
utils/env_frees.c \
utils/parsing_errors.c \
utils/execution_errors.c \
utils/debug.c

OBJECTS = $(CFILES:.c=.o)

LIBFT_PATH = libft
LIBFT_LIB = $(LIBFT_PATH)/libft.a
LIBFT_FLAGS = -L$(LIBFT_PATH) -lft

CC = cc
CFLAGS = -Wall -Wextra -Werror -I./includes
CFLAGS += -g3

LINKFLAGS = -lreadline -L/includes

%.o : %.c
	$(CC) $(CFLAGS) -c -o $@ $<

all: subsystems $(NAME)

subsystems:
	@make -C $(LIBFT_PATH)

$(NAME): $(OBJECTS) $(LIBFT_LIB)
	$(CC) $(CFLAGS) $(OBJECTS) $(LIBFT_FLAGS) $(LINKFLAGS) -o $(NAME)

clean:
	make clean -C libft/
	rm -rf $(OBJECTS)

fclean: clean
	make fclean -C libft/
	rm -f $(NAME)

re: fclean all

.PHONY:	all clean fclean re
